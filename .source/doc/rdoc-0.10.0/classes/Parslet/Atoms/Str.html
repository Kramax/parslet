  def try(io)
    old_pos = io.pos
    s = io.read(str.size)
    error(io, "Premature end of input") unless s && s.size==str.size
    error(io, "Expected #{str.inspect}, but got #{s.inspect}", old_pos) \
      unless s==str
    return s
  end
