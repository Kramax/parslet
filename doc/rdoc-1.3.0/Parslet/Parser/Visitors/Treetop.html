def visit_alternative(alternatives)
  '(' <<
  alternatives.
    map { |el| el.accept(self) }.
    join(' / ') <<
  ')'
end

def visit_repetition(tag, min, max, parslet)
  parslet.accept(self) << "#{min}..#{max}"
end
